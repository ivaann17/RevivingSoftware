126 	public static void infoCurso(CursoPropio cursoSeleccionado) {
127 		PantallaVisualizarCurso.id.setText(Integer.toString(cursoSeleccionado.getId()));
128 		PantallaVisualizarCurso.dniProf.setText(cursoSeleccionado.getDniDirector());
129 		PantallaVisualizarCurso.dniSec.setText(cursoSeleccionado.getDniSecretario());
130 		PantallaVisualizarCurso.edicion.setText(Integer.toString(cursoSeleccionado.getEdicion()));
131 		PantallaVisualizarCurso.nombreCurso.setText(cursoSeleccionado.getNombre());
132 		PantallaVisualizarCurso.numCreditos.setText(Integer.toString(cursoSeleccionado.getECTS()));
133 		PantallaVisualizarCurso.facultad.setText(cursoSeleccionado.getCentro());
134 		PantallaVisualizarCurso.precio.setText(Double.toString(cursoSeleccionado.getTasaMatricula()));
135 		PantallaVisualizarCurso.fechaIni.setText(cursoSeleccionado.getFechaInicio().toString());
136 		PantallaVisualizarCurso.fechaFin.setText(cursoSeleccionado.getFechaFin().toString());
137 		PantallaVisualizarCurso.mensaje = cursoSeleccionado.getMensaje();
138 	}
