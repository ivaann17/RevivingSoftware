205 	private CursoPropio obtenerCursoSeleccionado() {

209 	private void mostrarIngresosCurso() {
210 		try {
211 			double ingresosCurso = GestorConsultas.consultarIngresos(cursoSeleccionado);
212 			lblIngresosCurso.setVisible(true);
213 			ingresos.setVisible(true);
214 			ingresos.setText(Double.toString(ingresosCurso));
